def squares(n):
    """Compute the squares of numbers from 1 to n, such that the  
    ith element of the returned list equals i^2. 
     
    """
    # YOUR CODE HERE
    if n>=1:
        return [i**2 for i in range(1, n+1)]

    else:
        raise ValueError
